```typescript
// ─── Search hint normalization ────────────────────────────────────────────────
// Per Phase 9.1 D-LOCKED-Hint-Normalization: drop-on-invalid, never coerce,
// never throw. The whole hints object is dropped (returns null) only when the
// input is not a record. Field-level invalidity yields null/[] for that field.
// Accept BOTH raw LLM keys (`include_ingredients`, `max_ready_time_min`, etc.)
// and persisted normalized keys (`includeIngredients`, `maxReadyTime`, etc.).

export type SearchHints = {
  cuisine: string | null;
  type: string | null;
  includeIngredients: string[];
  excludeIngredients: string[];
  maxReadyTime: number | null;
  mainIngredient: string | null;
};

function isRecord(value: unknown): value is Record<string, unknown> {
  return typeof value === "object" && value !== null && !Array.isArray(value);
}

export function normalizeSearchHints(raw: unknown): SearchHints | null {
  if (!isRecord(raw)) return null;

  const cuisineRaw = typeof raw.cuisine === "string" ? raw.cuisine.trim().toLowerCase() : "";
  const typeRaw = typeof raw.type === "string" ? raw.type.trim().toLowerCase() : "";

  const includeSource = Array.isArray(raw.include_ingredients)
    ? raw.include_ingredients
    : Array.isArray(raw.includeIngredients)
      ? raw.includeIngredients
      : [];

  const includeIngredients = includeSource
    .filter((s): s is string => typeof s === "string" && s.trim().length > 0)
    .map((s) => s.trim().toLowerCase())
    .slice(0, 4);

  const excludeSource = Array.isArray(raw.exclude_ingredients)
    ? raw.exclude_ingredients
    : Array.isArray(raw.excludeIngredients)
      ? raw.excludeIngredients
      : [];

  const excludeIngredients = excludeSource
    .filter((s): s is string => typeof s === "string" && s.trim().length > 0)
    .map((s) => s.trim().toLowerCase())
    .slice(0, 8);

  const maxReadyTime =
    typeof raw.max_ready_time_min === "number" && raw.max_ready_time_min > 0
      ? Math.min(120, Math.max(5, Math.trunc(raw.max_ready_time_min)))
      : typeof raw.maxReadyTime === "number" && raw.maxReadyTime > 0
        ? Math.min(120, Math.max(5, Math.trunc(raw.maxReadyTime)))
      : null;

  const mainIngredient =
    typeof raw.main_ingredient === "string" && raw.main_ingredient.trim().length > 0
      ? raw.main_ingredient.trim().toLowerCase()
      : typeof raw.mainIngredient === "string" && raw.mainIngredient.trim().length > 0
        ? raw.mainIngredient.trim().toLowerCase()
      : null;

  return {
    cuisine: CUISINE_ENUM.has(cuisineRaw) ? cuisineRaw : null,
    type: TYPE_ENUM.has(typeRaw) ? typeRaw : null,
    includeIngredients,
    excludeIngredients,
    maxReadyTime,
    mainIngredient,
  };
}
```
