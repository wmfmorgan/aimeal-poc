KPI 1: First-call hit rate
- Numerator: count(spoonacular_usage where cache_hit=false AND endpoint = 'recipes/complexSearch?addRecipeInformation')
- Denominator: count(spoonacular_usage where cache_hit=false AND endpoint like 'recipes/complexSearch?addRecipeInformation%')

KPI 2: Enum drop rate
- During normalization, every invalid non-empty `cuisine` or `type` value emits exactly one structured drop marker into the same `llm_logs.response` record that captured the original LLM output
- Dev diagnostics scan the most recent N=25 `llm_logs.response` entries for those structured marker lines, not the raw NDJSON payload
- Count: total emitted = number of non-empty cuisine/type values observed during normalization; dropped = number of emitted values that produced a structured drop marker
- Aggregate across all scanned llm_logs rows
- Rate = dropped / totalEmitted

KPI 3: Empty instructions rate
- Numerator: count(spoonacular_cache where instructions is null OR jsonb_array_length(coalesce(instructions, '[]'::jsonb)) = 0)
  (Actually `instructions` is `text[]`, not jsonb — verify against migration `20260419000001_initial_schema.sql`. If text[], use `coalesce(array_length(instructions, 1), 0) = 0`.)
- Denominator: count(spoonacular_cache)
- Rate = numerator / denominator

### Sub-step A: Append `countLoggedEnumDrops` to `supabase/functions/_shared/spoonacular-mappings.ts`

```typescript
/**
 * Phase 9.1 D-LOCKED-Telemetry-KPIs (KPI 2 — Enum drop rate).
 *
 * Plan 02 normalization already emits one structured log marker per invalid
 * non-empty cuisine/type candidate plus a compact summary footer. This helper
 * only reads those logged markers so dev diagnostics can compute emitted vs
 * dropped counts without reparsing the meal NDJSON.
 */
export function countLoggedEnumDrops(
  responseText: string,
): { totalEmitted: number; dropped: number } {
  if (!responseText) return { totalEmitted: 0, dropped: 0 };

  const summaryMatch = responseText.match(/\[\[enum-drop-summary:emitted=(\d+);dropped=(\d+)\]\]/);
  if (!summaryMatch) return { totalEmitted: 0, dropped: 0 };

  return {
    totalEmitted: Number(summaryMatch[1] ?? 0),
    dropped: Number(summaryMatch[2] ?? 0),
  };
}
```

Plan 02 fully owns marker emission in `generate-draft/index.ts` and `trpc/index.ts`. Do not reschedule those producer edits here; this plan only consumes the marker lines already persisted into `llm_logs.response`.

```typescript
spoonacularUsage: authedProcedure.query(async ({ ctx }) => {
  const todayUtc = new Date().toISOString().slice(0, 10);
  const { data, error } = await ctx.supabase
    .from("spoonacular_usage")
    .select(
      "meal_id, meal_plan_id, spoonacular_recipe_id, cache_hit, endpoint, points_used, quota_request, quota_used, quota_left, usage_date_utc, created_at"
    )
    .order("created_at", { ascending: false })
    .limit(25);

  if (error) {
    throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: error.message });
  }

  const entries = (data ?? []).map((row) => buildUsageEvent(row));
  const summaries = summarizeUsageByUtcDay(entries, SPOONACULAR_DAILY_LIMIT);

  // ─── Phase 9.1 D-LOCKED-Telemetry-KPIs ────────────────────────────────────

  // KPI 1: First-call hit rate.
  // Numerator: live (non-cache) calls with the base first-attempt-success endpoint.
  // Denominator: all live (non-cache) calls, including retry successes and retry no-result failures.
  const { data: liveCalls, error: liveError } = await ctx.supabase
    .from("spoonacular_usage")
    .select("endpoint")
    .eq("cache_hit", false);
  if (liveError) {
    throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: liveError.message });
  }
  const firstCallDenominator = (liveCalls ?? []).length;
  const firstCallHits = (liveCalls ?? []).filter(
    (row) => row.endpoint === "recipes/complexSearch?addRecipeInformation",
  ).length;
  const firstCallHitRate =
    firstCallDenominator === 0 ? 0 : firstCallHits / firstCallDenominator;

  // KPI 2: Enum drop rate (last 25 LLM responses with marker lines appended during normalization).
  const { data: recentLlmLogs, error: llmError } = await ctx.supabase
    .from("llm_logs")
    .select("response")
    .order("created_at", { ascending: false })
    .limit(25);
  if (llmError) {
    throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: llmError.message });
  }
  let enumDropDenominator = 0;
  let enumDropNumerator = 0;
  for (const row of recentLlmLogs ?? []) {
    const text = typeof row.response === "string" ? row.response : "";
    const counts = countLoggedEnumDrops(text);
    enumDropDenominator += counts.totalEmitted;
    enumDropNumerator += counts.dropped;
  }
  const enumDropRate =
    enumDropDenominator === 0 ? 0 : enumDropNumerator / enumDropDenominator;

  // KPI 3: Empty instructions rate (entire cache).
  // Branch this depending on what the column type actually is. Default below
  // assumes `text[]`. Adjust if Sub-step A reveals jsonb.
  const { data: cachedRecipes, error: cacheError } = await ctx.serviceSupabase
    .from("spoonacular_cache")
    .select("spoonacular_recipe_id, instructions");
  if (cacheError) {
    throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: cacheError.message });
  }
  const emptyInstructionsDenominator = (cachedRecipes ?? []).length;
  const emptyInstructionsNumerator = (cachedRecipes ?? []).filter((row) => {
    const instructions = row.instructions;
    if (instructions == null) return true;
    if (Array.isArray(instructions) && instructions.length === 0) return true;
    return false;
  }).length;
  const emptyInstructionsRate =
    emptyInstructionsDenominator === 0
      ? 0
      : emptyInstructionsNumerator / emptyInstructionsDenominator;

  return {
    today: summaries.find((summary) => summary.usage_date_utc === todayUtc) ?? {
      usage_date_utc: todayUtc,
      requests_made: 0,
      cache_hits: 0,
      cache_misses: 0,
      points_used: 0,
      quota_used: 0,
      quota_left: SPOONACULAR_DAILY_LIMIT,
      daily_limit: SPOONACULAR_DAILY_LIMIT,
    },
    recent: entries,
    liveConcurrencyLimit: SPOONACULAR_LIVE_CONCURRENCY_LIMIT,
    kpis: {
      firstCallHitRate,
      firstCallDenominator,
      enumDropRate,
      enumDropDenominator,
      emptyInstructionsRate,
      emptyInstructionsDenominator,
    },
  };
}),
```
