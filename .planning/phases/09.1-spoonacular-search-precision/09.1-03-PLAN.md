### Sub-step B: Add the `searchAndEnrich` helper just BELOW the existing `extractBlockedTerm` function (around current line 425, before `insertUsageEvent`)

Embed this code verbatim:

```typescript
type SpoonacularComplexSearchResult = SpoonacularRecipePayload & {
  id: number;
  title?: string;
};

/**
 * Phase 9.1 D-LOCKED-Enrichment-Consolidation.
 *
 * Single-network-round-trip recipe lookup. Calls /recipes/complexSearch with
 * addRecipeInformation+addRecipeNutrition so the response contains full
 * extendedIngredients, nutrition, analyzedInstructions inline.
 *
 * Hard filters (diet, intolerances, excludeIngredients) are ALWAYS present.
 * Soft filters (equipment, maxReadyTime) are dropped on retry.
 * No further widening — return null if both attempts fail.
 */
async function searchAndEnrich(
  meal: { title: string; meal_type: string; search_hints: SearchHints | null },
  household: HouseholdRestrictions,
  retryWithRelaxedFilters = false,
): Promise<
  | { recipe: SpoonacularComplexSearchResult; response: Awaited<ReturnType<typeof fetchSpoonacularJson>>; searchEndpoint: string }
  | { recipe: null; response: Awaited<ReturnType<typeof fetchSpoonacularJson>>; searchEndpoint: string }
> {
  const hints = meal.search_hints;
  const params: Record<string, string> = {
    query: meal.title,
    number: "1",
    instructionsRequired: "true",
    sort: "popularity",
    addRecipeInformation: "true",
    addRecipeNutrition: "true",
  };

  // Hard filters — never dropped, even on retry.
  if (household.diet) params.diet = household.diet;
  if (household.intolerances.length > 0) {
    params.intolerances = household.intolerances.join(",");
  }
  const excludes = [
    ...new Set([
      ...household.excludeIngredients,
      ...(hints?.excludeIngredients ?? []),
    ]),
  ];
  if (excludes.length > 0) params.excludeIngredients = excludes.join(",");

  // LLM-derived recipe-specific filters — kept on retry too.
  if (hints?.includeIngredients?.length) {
    params.includeIngredients = hints.includeIngredients.join(",");
  }
  if (hints?.cuisine) params.cuisine = hints.cuisine;

  const typeValue = hints?.type ?? mealTypeToSpoonacularType(meal.meal_type);
  if (typeValue) params.type = typeValue;

  // Soft filters — dropped on retry.
  if (!retryWithRelaxedFilters) {
    if (household.equipment.length > 0) {
      params.equipment = household.equipment.join(",");
    }
    const timeCap =
      hints?.maxReadyTime != null && household.maxReadyTime != null
        ? Math.min(hints.maxReadyTime, household.maxReadyTime)
        : (hints?.maxReadyTime ?? household.maxReadyTime);
    if (timeCap != null) params.maxReadyTime = String(timeCap);
  }

  const response = await fetchSpoonacularJson("/recipes/complexSearch", params);
  const result = response.json?.results?.[0] as SpoonacularComplexSearchResult | undefined;

  if (!result?.id) {
    if (retryWithRelaxedFilters) {
      return {
        recipe: null,
        response,
        searchEndpoint: "recipes/complexSearch?addRecipeInformation&retryRelaxed&noResult",
      };
    }
    return searchAndEnrich(meal, household, true);
  }

  return {
    recipe: result,
    response,
    searchEndpoint: retryWithRelaxedFilters
      ? "recipes/complexSearch?addRecipeInformation&retryRelaxed"
      : "recipes/complexSearch?addRecipeInformation",
  };
}
```
