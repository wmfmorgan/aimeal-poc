```typescript
/**
 * Spoonacular vocabulary and household restriction aggregation.
 *
 * Source of truth for:
 *  - Phase 9.1 D-LOCKED-Enum-Tables (cuisine/type/diet/intolerance/equipment)
 *  - Phase 9.1 D-LOCKED-Household-Aggregation-Rules
 *
 * Imported by both `generate-draft/index.ts` (writeMealToDB hint persistence)
 * and `trpc/index.ts` (regenerate-meal hint persistence + meal.enrich
 * consolidated search). Keep this file Deno-compatible: no Node imports,
 * no runtime deps. Pure data + pure functions only.
 */

export const CUISINE_ENUM: ReadonlySet<string> = new Set([
  "african","american","british","cajun","caribbean","chinese","eastern european",
  "european","french","german","greek","indian","irish","italian","japanese","jewish","korean",
  "latin american","mediterranean","mexican","middle eastern","nordic","southern","spanish","thai","vietnamese",
]);

export const TYPE_ENUM: ReadonlySet<string> = new Set([
  "main course","side dish","dessert","appetizer","salad","bread","breakfast","soup",
  "beverage","sauce","marinade","fingerfood","snack","drink",
]);

export const DIET_ENUM: ReadonlySet<string> = new Set([
  "gluten free","ketogenic","vegetarian","lacto-vegetarian","ovo-vegetarian","vegan",
  "pescetarian","paleo","primal","low fodmap","whole30",
]);

export const INTOLERANCE_ENUM: ReadonlySet<string> = new Set([
  "dairy","egg","gluten","grain","peanut","seafood","sesame","shellfish","soy","sulfite",
  "tree nut","wheat",
]);

// Equipment enum is large (~50 values) in Spoonacular. Hardcode the subset the
// UI dropdown actually offers. Append values here as the UI dropdown grows.
export const EQUIPMENT_ENUM: ReadonlySet<string> = new Set([
  "blender","oven","frying pan","slow cooker","pressure cooker","grill","microwave",
  "food processor","stand mixer","immersion blender","wok","dutch oven",
]);

const MEAL_TYPE_TO_SPOONACULAR: Record<string, string> = {
  breakfast: "breakfast",
  lunch: "main course",
  dinner: "main course",
};

export function mealTypeToSpoonacularType(mealType: string): string | null {
  return MEAL_TYPE_TO_SPOONACULAR[mealType] ?? null;
}

export type Member = {
  allergies: string[];
  avoidances: string[];
  diet_type: string | null;
};

export type HouseholdRestrictions = {
  diet: string | null;          // single enum value when all members agree, else null
  intolerances: string[];       // enum values, union across members
  excludeIngredients: string[]; // free-text avoidances + non-enum allergies
  equipment: string[];          // enum values from household.appliances
  maxReadyTime: number | null;  // derived from skill level
};

const SKILL_TIME_CAP: Record<string, number | null> = {
  beginner: 30,
  intermediate: 60,
  advanced: null,
};

export function aggregateHouseholdRestrictions(
  members: Member[],
  appliances: string[],
  skillLevel: string,
): HouseholdRestrictions {
  const intolerances = new Set<string>();
  const excludeIngredients = new Set<string>();
  const memberDiets = new Set<string>();

  for (const m of members) {
    for (const a of m.allergies) {
      const lower = a.trim().toLowerCase();
      if (!lower) continue;
      if (INTOLERANCE_ENUM.has(lower)) intolerances.add(lower);
      else excludeIngredients.add(lower);
    }
    for (const a of m.avoidances) {
      const lower = a.trim().toLowerCase();
      if (lower) excludeIngredients.add(lower);
    }
    if (m.diet_type) {
      const lower = m.diet_type.trim().toLowerCase();
      if (DIET_ENUM.has(lower)) memberDiets.add(lower);
    }
  }

  const diet = memberDiets.size === 1 ? [...memberDiets][0] : null;
  const equipment = appliances
    .map((a) => a.trim().toLowerCase())
    .filter((a) => EQUIPMENT_ENUM.has(a));

  return {
    diet,
    intolerances: [...intolerances],
    excludeIngredients: [...excludeIngredients],
    equipment,
    maxReadyTime: SKILL_TIME_CAP[skillLevel] ?? null,
  };
}
```

```typescript
import { describe, expect, it } from "vitest";

import {
  CUISINE_ENUM,
  DIET_ENUM,
  EQUIPMENT_ENUM,
  INTOLERANCE_ENUM,
  TYPE_ENUM,
  aggregateHouseholdRestrictions,
  mealTypeToSpoonacularType,

describe("Spoonacular enum tables", () => {
  it("contains every locked cuisine value (sample anchors)", () => {
    expect(CUISINE_ENUM.has("african")).toBe(true);
    expect(CUISINE_ENUM.has("eastern european")).toBe(true);
    expect(CUISINE_ENUM.has("mediterranean")).toBe(true);
    expect(CUISINE_ENUM.has("vietnamese")).toBe(true);
    expect(CUISINE_ENUM.size).toBe(26);
  });

  it("contains every locked type value (sample anchors)", () => {
    expect(TYPE_ENUM.has("main course")).toBe(true);
    expect(TYPE_ENUM.has("breakfast")).toBe(true);
    expect(TYPE_ENUM.has("drink")).toBe(true);
    expect(TYPE_ENUM.size).toBe(14);
  });

  it("contains every locked diet value (sample anchors)", () => {
    expect(DIET_ENUM.has("gluten free")).toBe(true);
    expect(DIET_ENUM.has("vegetarian")).toBe(true);
    expect(DIET_ENUM.has("whole30")).toBe(true);
    expect(DIET_ENUM.size).toBe(11);
  });

  it("contains every locked intolerance value", () => {
    expect(INTOLERANCE_ENUM.has("dairy")).toBe(true);
    expect(INTOLERANCE_ENUM.has("tree nut")).toBe(true);
    expect(INTOLERANCE_ENUM.has("wheat")).toBe(true);
    expect(INTOLERANCE_ENUM.size).toBe(12);
  });

  it("contains the curated equipment subset", () => {
    expect(EQUIPMENT_ENUM.has("blender")).toBe(true);
    expect(EQUIPMENT_ENUM.has("dutch oven")).toBe(true);
    expect(EQUIPMENT_ENUM.has("toaster oven")).toBe(false); // intentionally excluded
  });
});

describe("mealTypeToSpoonacularType", () => {
  it("maps breakfast/lunch/dinner correctly and returns null for unknowns", () => {
    expect(mealTypeToSpoonacularType("breakfast")).toBe("breakfast");
    expect(mealTypeToSpoonacularType("lunch")).toBe("main course");
    expect(mealTypeToSpoonacularType("dinner")).toBe("main course");
    expect(mealTypeToSpoonacularType("snack")).toBeNull();
    expect(mealTypeToSpoonacularType("")).toBeNull();
  });
});

describe("aggregateHouseholdRestrictions", () => {
  it("routes enum allergies to intolerances and non-enum allergies to excludeIngredients", () => {
    const result = aggregateHouseholdRestrictions(
      [{ allergies: ["dairy", "Strawberry"], avoidances: [], diet_type: null }],
      [],
      "intermediate",
    );
    expect(result.intolerances).toEqual(["dairy"]);
    expect(result.excludeIngredients).toContain("strawberry");
  });

  it("returns single diet when all members agree", () => {
    const result = aggregateHouseholdRestrictions(
      [
        { allergies: [], avoidances: [], diet_type: "vegetarian" },
        { allergies: [], avoidances: [], diet_type: "Vegetarian" },
      ],
      [],
      "intermediate",
    );
    expect(result.diet).toBe("vegetarian");
  });

  it("returns null diet when members disagree", () => {
    const result = aggregateHouseholdRestrictions(
      [
        { allergies: [], avoidances: [], diet_type: "vegetarian" },
        { allergies: [], avoidances: [], diet_type: "vegan" },
      ],
      [],
      "intermediate",
    );
    expect(result.diet).toBeNull();
  });

  it("trims and lowercases allergies/avoidances; drops empties", () => {
    const result = aggregateHouseholdRestrictions(
      [
        { allergies: [" Dairy "], avoidances: ["  ", "Cilantro"], diet_type: null },
      ],
      [],
      "intermediate",
    );
    expect(result.intolerances).toEqual(["dairy"]);
    expect(result.excludeIngredients).toEqual(["cilantro"]);
  });

  it("filters appliances against EQUIPMENT_ENUM (drop-on-invalid)", () => {
    const result = aggregateHouseholdRestrictions(
      [],
      ["Blender", "ToasterOven", "microwave"],
      "intermediate",
    );
    expect(result.equipment).toEqual(["blender", "microwave"]);
  });

  it("derives maxReadyTime from skill level", () => {
    expect(
      aggregateHouseholdRestrictions([], [], "beginner").maxReadyTime,
    ).toBe(30);
    expect(
      aggregateHouseholdRestrictions([], [], "intermediate").maxReadyTime,
    ).toBe(60);
    expect(
      aggregateHouseholdRestrictions([], [], "advanced").maxReadyTime,
    ).toBeNull();
    expect(
      aggregateHouseholdRestrictions([], [], "").maxReadyTime,
    ).toBeNull();
  });

  it("handles fully empty inputs", () => {
    expect(aggregateHouseholdRestrictions([], [], "intermediate")).toEqual({
      diet: null,
      intolerances: [],
      excludeIngredients: [],
      equipment: [],
      maxReadyTime: 60,
    });
  });

  it("never relaxes hard restrictions across union members", () => {
    const result = aggregateHouseholdRestrictions(
      [
        { allergies: ["dairy"], avoidances: ["mushrooms"], diet_type: null },
        { allergies: ["peanut"], avoidances: ["cilantro"], diet_type: null },
      ],
      [],
      "advanced",
    );
    expect(result.intolerances.sort()).toEqual(["dairy", "peanut"]);
    expect(result.excludeIngredients.sort()).toEqual(["cilantro", "mushrooms"]);
  });
});
```
