```typescript
import OpenAI from "https://deno.land/x/openai@v4.69.0/mod.ts";
import { createClient } from "npm:@supabase/supabase-js@2";

// CRITICAL: Deno URL import for OpenAI SDK â€” NOT npm: specifier
// CRITICAL: model "grok-4-1-fast-non-reasoning" â€” NEVER grok-3-mini

const xaiClient = new OpenAI({
  apiKey: Deno.env.get("XAI_API_KEY") ?? "",
  baseURL: "https://api.x.ai/v1",
});

// Build the system prompt instructing NDJSON output â€” NO response_format (spike landmine: incompatible with stream:true)
function buildSystemPrompt(numDays: number, mealTypes: string[]): string {
  const totalMeals = numDays * mealTypes.length;
  const mealTypesList = mealTypes.join(", ");
  return `You are a registered dietitian and PhD nutritionist.
Output ONLY one JSON object per line â€” no markdown, no wrapper array, no explanation.
Each line must match this exact shape:
{"day_of_week":"Monday","meal_type":"breakfast","title":"Meal title here","short_description":"One sentence description here."}

Days: Monday, Tuesday, Wednesday, Thursday, Friday, Saturday, Sunday (use first ${numDays} days only).
Meal types per day: ${mealTypesList}.
Output exactly ${totalMeals} lines, one per meal slot. No extra lines.`;
}

function buildUserPrompt(config: {
  householdName: string;
  skillLevel: string;
  members: Array<{ name: string; allergies: string[]; avoidances: string[]; diet_type?: string | null }>;
  appliances: string[];
  numDays: number;
  mealTypes: string[];
}): string {
  const memberSummary = config.members
    .map(
      (m) =>
        `- ${m.name}: allergies=[${m.allergies.join(", ")}], avoids=[${m.avoidances.join(", ")}]${m.diet_type ? `, diet=${m.diet_type}` : ""}`
    )
    .join("\n");

  const days = ["Monday","Tuesday","Wednesday","Thursday","Friday","Saturday","Sunday"].slice(0, config.numDays);

  return `Household: ${config.householdName}
Cooking skill: ${config.skillLevel}
Appliances: ${config.appliances.join(", ") || "standard kitchen"}
Members:
${memberSummary}

Generate a ${config.numDays}-day meal plan (${days.join(", ")}) with ${config.mealTypes.join(" + ")} for each day.
Respect ALL listed allergies and avoidances. Match the cooking skill level and available appliances.`;
}

function normalizeArray(value: unknown): string[] {
  return Array.isArray(value) ? value.filter((v): v is string => typeof v === "string") : [];
}

Deno.serve(async (req) => {
  // CORS preflight
  if (req.method === "OPTIONS") {
    return new Response(null, {
      headers: {
        "Access-Control-Allow-Origin": "*",
        "Access-Control-Allow-Headers": "authorization, content-type",
        "Access-Control-Allow-Methods": "POST, OPTIONS",
      },
    });
  }

  const supabaseUrl = Deno.env.get("SUPABASE_URL") ?? "";
  const supabaseAnonKey = Deno.env.get("SUPABASE_ANON_KEY") ?? "";
  const authHeader = req.headers.get("Authorization") ?? "";

  // 1. Auth validation â€” replicate trpc/index.ts pattern exactly
  const supabase = createClient(supabaseUrl, supabaseAnonKey, {
    global: { headers: authHeader ? { Authorization: authHeader } : {} },
  });

  let userId: string | null = null;
  if (authHeader.startsWith("Bearer ")) {
    const token = authHeader.slice("Bearer ".length);
    const { data } = await supabase.auth.getUser(token);
    userId = data.user?.id ?? null;
  }
  if (!userId) {
    return new Response("Unauthorized", { status: 401 });
  }

  // 2. Parse request body
  let householdId: string;
  let mealPlanId: string;
  let numDays: number;
  let mealTypes: string[];

  try {
    const body = await req.json();
    householdId = body.householdId;
    mealPlanId = body.mealPlanId;
    numDays = typeof body.numDays === "number" ? body.numDays : 7;
    mealTypes = Array.isArray(body.mealTypes) ? body.mealTypes : ["breakfast", "lunch", "dinner"];
  } catch {
    return new Response("Invalid request body", { status: 400 });
  }

  if (!householdId || !mealPlanId) {
    return new Response("householdId and mealPlanId are required", { status: 400 });
  }

  // 3. Fetch household from DB (scoped to caller's JWT â€” RLS enforces ownership)
  const { data: household, error: householdError } = await supabase
    .from("households")
    .select(
      `id, name, cooking_skill_level, appliances,
       household_members ( id, name, allergies, avoidances, diet_type )`
    )
    .eq("id", householdId)
    .single();

  if (householdError || !household) {
    return new Response("Household not found", { status: 404 });
  }

  const members = Array.isArray(household.household_members) ? household.household_members : [];

  const systemPrompt = buildSystemPrompt(numDays, mealTypes);
  const userPrompt = buildUserPrompt({
    householdName: household.name,
    skillLevel: household.cooking_skill_level ?? "intermediate",
    members: members.map((m: { name: string; allergies: unknown; avoidances: unknown; diet_type?: string | null }) => ({
      name: m.name,
      allergies: normalizeArray(m.allergies),
      avoidances: normalizeArray(m.avoidances),
      diet_type: m.diet_type ?? null,
    })),
    appliances: normalizeArray(household.appliances),
    numDays,
    mealTypes,
  });

  // 4. Streaming LLM call â€” CRITICAL: NO response_format (incompatible with stream:true, spike-validated landmine)
  const llmStream = await xaiClient.chat.completions.create({
    model: "grok-4-1-fast-non-reasoning",
    temperature: 0.7,
    stream: true,
    messages: [
      { role: "system", content: systemPrompt },
      { role: "user", content: userPrompt },
    ],
  });

  // 5. ReadableStream: buffer tokens, emit SSE per completed NDJSON line, write meals to DB
  let tokenBuffer = "";
  let fullResponse = "";
  let promptTokens = 0;
  let completionTokens = 0;

  const sseStream = new ReadableStream({
    async start(controller) {
      const encoder = new TextEncoder();

      function writeMealToDb(jsonLine: string): void {
        // Fire-and-forget â€” do NOT await (would block SSE emission)
        try {
          const meal = JSON.parse(jsonLine);
          supabase
            .from("meals")
            .insert({
              meal_plan_id: mealPlanId,
              day_of_week: meal.day_of_week,
              meal_type: meal.meal_type,
              title: meal.title,
              short_description: meal.short_description ?? null,
              status: "draft",
            })
            .then(() => { /* intentional no-op */ });
        } catch {
          // Silently skip malformed lines â€” already validated above
        }
      }

      try {
        for await (const chunk of llmStream) {
          const token = chunk.choices[0]?.delta?.content ?? "";
          tokenBuffer += token;
          fullResponse += token;

          // Detect completed NDJSON lines by newline character
          const lines = tokenBuffer.split("\n");
          tokenBuffer = lines.pop() ?? ""; // last element is incomplete; keep in buffer

          for (const line of lines) {
            const trimmed = line.trim();
            if (!trimmed) continue;
            try {
              JSON.parse(trimmed); // validate JSON before emitting
              controller.enqueue(encoder.encode(`data: ${trimmed}\n\n`));
              writeMealToDb(trimmed);
            } catch {
              // Malformed line from Grok â€” skip silently (Pitfall 3)
              console.error("[generate-draft] malformed NDJSON line skipped:", trimmed.slice(0, 80));
            }
          }

          // Capture usage from final chunk (Grok sends usage on last chunk)
          if (chunk.usage) {
            promptTokens = chunk.usage.prompt_tokens ?? 0;
            completionTokens = chunk.usage.completion_tokens ?? 0;
          }
        }

        // Handle any remaining buffer after stream ends
        const remaining = tokenBuffer.trim();
        if (remaining) {
          try {
            JSON.parse(remaining);
            controller.enqueue(encoder.encode(`data: ${remaining}\n\n`));
            writeMealToDb(remaining);
          } catch { /* skip */ }
        }
      } finally {
        // 6. Persist LLM log after stream completes (or on error)
        await supabase.from("llm_logs").insert({
          household_id: householdId,
          model: "grok-4-1-fast-non-reasoning",
          prompt: userPrompt,
          response: fullResponse,
          prompt_tokens: promptTokens || null,
          completion_tokens: completionTokens || null,
        });

        // 7. Terminate stream
        controller.enqueue(encoder.encode("data: [DONE]\n\n"));
        controller.close();
      }
    },
  });

  return new Response(sseStream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      "Access-Control-Allow-Origin": "*",
    },
  });
});
```

| T-04-02-01 | Elevation of Privilege | generate-draft POST handler | mitigate | JWT validated at function entry via `supabase.auth.getUser(token)`; 401 returned before any LLM call if missing/invalid |
| T-04-02-02 | Tampering | householdId in POST body | mitigate | Edge function fetches household scoped to caller's JWT (`supabase.from('households').select().eq('id', householdId).single()` â€” RLS `households: own rows` restricts to `auth.uid() = user_id`); attacker cannot read another user's household |
| T-04-02-03 | Tampering | mealPlanId in POST body | mitigate | `meals` insert uses `meal_plan_id` FK; RLS `meals: own meal plans` restricts inserts to plans owned by `auth.uid()` â€” attacker cannot inject meals into another user's plan |

| T-04-02-05 | Denial of Service | Unbounded LLM generation requests | accept | No rate limiting in Phase 4 PoC. Risk is low (requires valid auth token; cost is borne by API key owner). Future: rate limit per userId |
| T-04-02-06 | Information Disclosure | XAI_API_KEY | mitigate | Key loaded from `Deno.env.get("XAI_API_KEY")` â€” never serialized to response body; not logged |
