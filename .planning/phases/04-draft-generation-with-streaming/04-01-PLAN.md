```typescript
import { type MealSlot, DAYS_OF_WEEK } from "./types";

export function parseMealLine(line: string): MealSlot | null {
  try {
    const obj = JSON.parse(line);
    if (
      typeof obj.day_of_week === "string" &&
      typeof obj.meal_type === "string" &&
      typeof obj.title === "string" &&
      typeof obj.short_description === "string"
    ) {
      return obj as MealSlot;
    }
  } catch {
    // ignore malformed lines
  }
  return null;
}

export function buildSlotKey(day: string, mealType: string): string {
  return `${day}::${mealType}`;
}

export function buildExpectedSlots(numDays: number, mealTypes: string[]): string[] {
  const days = DAYS_OF_WEEK.slice(0, numDays);
  return days.flatMap((day) => mealTypes.map((type) => buildSlotKey(day, type)));
}
```

```typescript
import { describe, it, expect } from "vitest";
import { parseMealLine, buildSlotKey, buildExpectedSlots } from "./stream-parser";

describe("parseMealLine", () => {
  it("returns a MealSlot for a valid NDJSON line", () => {
    const line = JSON.stringify({
      day_of_week: "Monday",
      meal_type: "breakfast",
      title: "Oat Porridge",
      short_description: "A warming bowl.",
    });
    const result = parseMealLine(line);
    expect(result).not.toBeNull();
    expect(result?.day_of_week).toBe("Monday");
    expect(result?.meal_type).toBe("breakfast");
    expect(result?.title).toBe("Oat Porridge");
  });

  it("returns null for malformed JSON", () => {
    expect(parseMealLine("{not valid json")).toBeNull();
  });

  it("returns null when a required field is missing", () => {
    const line = JSON.stringify({ day_of_week: "Monday", meal_type: "lunch" });
    expect(parseMealLine(line)).toBeNull();
  });
});

describe("buildSlotKey", () => {
  it("returns day::mealType format", () => {
    expect(buildSlotKey("Monday", "breakfast")).toBe("Monday::breakfast");
  });
});

describe("buildExpectedSlots", () => {
  it("returns numDays * mealTypes.length keys", () => {
    const slots = buildExpectedSlots(3, ["lunch", "dinner"]);
    expect(slots).toHaveLength(6);
    expect(slots[0]).toBe("Monday::lunch");
    expect(slots[1]).toBe("Monday::dinner");
    expect(slots[2]).toBe("Tuesday::lunch");
  });

  it("respects numDays cap (7 max from DAYS_OF_WEEK)", () => {
    const slots = buildExpectedSlots(7, ["dinner"]);
    expect(slots).toHaveLength(7);
    expect(slots[6]).toBe("Sunday::dinner");
  });
});
```
