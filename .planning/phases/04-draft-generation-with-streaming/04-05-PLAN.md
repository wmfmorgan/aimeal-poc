**`tests/e2e/generation-error.spec.ts`** (replace test.todo() stubs):
```typescript
import { test, expect } from "@playwright/test";

test.describe("Generation stream error", () => {
  test.beforeEach(async ({ page }) => {
    // Mock SSE endpoint to return a 500 error
    await page.route("**/functions/v1/generate-draft", async (route) => {
      await route.fulfill({
        status: 500,
        body: "Internal Server Error",
      });
    });

    // Mock the mealPlan.create mutation
    await page.route("**/trpc/mealPlan.create**", async (route) => {
      await route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify({ result: { data: { id: "test-plan-id-error" } } }),
      });
    });

    await page.goto("/plan/new");
  });

  test("Stream error: StreamErrorBanner is shown after failed generation", async ({ page }) => {
    await page.getByRole("button", { name: /Generate Your Plan/i }).click();
    await expect(page.getByRole("alert").filter({ hasText: /wrong during generation/i })).toBeVisible({
      timeout: 5000,
    });
  });

  test("Stream error: 'Try again →' button is visible in the error banner", async ({ page }) => {
    await page.getByRole("button", { name: /Generate Your Plan/i }).click();
    await expect(page.getByRole("button", { name: /Try again/i })).toBeVisible({ timeout: 5000 });
  });

  test("Stream error: clicking 'Try again →' resets to idle/form state", async ({ page }) => {
    await page.getByRole("button", { name: /Generate Your Plan/i }).click();
    const retryBtn = page.getByRole("button", { name: /Try again/i });
    await retryBtn.waitFor({ timeout: 5000 });
    await retryBtn.click();
    // After retry, form should be visible again
    await expect(page.getByRole("button", { name: /Generate Your Plan/i })).toBeVisible({ timeout: 2000 });
  });
});
```
