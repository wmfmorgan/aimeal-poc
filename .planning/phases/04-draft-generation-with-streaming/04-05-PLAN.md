import { test, expect } from "@playwright/test";

// Mock SSE meal data — 3 meals for "Dinner only, 3 days" to keep tests fast
const MOCK_MEALS = [
  { day_of_week: "Monday", meal_type: "dinner", title: "Roast Chicken", short_description: "A classic roast." },
  { day_of_week: "Tuesday", meal_type: "dinner", title: "Pasta Bolognese", short_description: "Rich meat sauce." },
  { day_of_week: "Wednesday", meal_type: "dinner", title: "Grilled Salmon", short_description: "Light and healthy." },
];

// Helper: build a mock SSE response body from meal data
function buildMockSSEBody(meals: typeof MOCK_MEALS): string {
  const lines = meals.map((m) => `data: ${JSON.stringify(m)}`).join("\n\n");
  return `${lines}\n\ndata: [DONE]\n\n`;
}

test.describe("Generation flow", () => {
  test.beforeEach(async ({ page }) => {
    // Intercept the generate-draft edge function with a mock SSE stream
    await page.route("**/functions/v1/generate-draft", async (route) => {
      await route.fulfill({
        status: 200,
        headers: {
          "Content-Type": "text/event-stream",
          "Cache-Control": "no-cache",
          "Access-Control-Allow-Origin": "*",
        },
        body: buildMockSSEBody(MOCK_MEALS),
      });
    });

    // Intercept mealPlan.create tRPC call to return a fake plan ID
    await page.route("**/trpc/mealPlan.create*", async (route) => {
      await route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify({ result: { data: { id: "test-plan-uuid-123" } } }),
      });
    });

    // Navigate to /plan/new (requires auth — check if auth is needed)
    await page.goto("/plan/new");
  });

  test("form renders with preset buttons and day count stepper", async ({ page }) => {
    await expect(page.getByText("Dinner only")).toBeVisible();
    await expect(page.getByText("Lunch + Dinner")).toBeVisible();
    await expect(page.getByText("All three")).toBeVisible();
    await expect(page.getByText("7")).toBeVisible(); // default day count
  });

  test("selecting Dinner only preset and submitting shows skeleton grid", async ({ page }) => {
    await page.getByText("Dinner only").click();
    // Reduce days for faster test
    for (let i = 0; i < 4; i++) {
      await page.getByLabel("Decrease days").click();
    }
    await expect(page.getByText("3")).toBeVisible();

    await page.getByText("Generate Your Plan →").click();

    // Skeleton grid should appear immediately
    await expect(page.getByLabel("Loading meal").first()).toBeVisible({ timeout: 3000 });
  });

  test("meal cards replace skeleton slots after stream completes", async ({ page }) => {
    await page.getByText("Dinner only").click();
    for (let i = 0; i < 4; i++) await page.getByLabel("Decrease days").click();
    await page.getByText("Generate Your Plan →").click();

    // Wait for a real meal card to appear
    await expect(page.getByText("Roast Chicken")).toBeVisible({ timeout: 5000 });
    await expect(page.getByText("Pasta Bolognese")).toBeVisible({ timeout: 5000 });
    await expect(page.getByText("Grilled Salmon")).toBeVisible({ timeout: 5000 });
  });

  test("PlanReadyBanner appears after stream complete", async ({ page }) => {
    await page.getByText("Dinner only").click();
    for (let i = 0; i < 4; i++) await page.getByLabel("Decrease days").click();
    await page.getByText("Generate Your Plan →").click();

    await expect(
      page.getByText("Your plan is ready. Meals will be editable in the next update.")
    ).toBeVisible({ timeout: 5000 });
  });

  test("URL updates to /plan/:id after mealPlan.create resolves", async ({ page }) => {
    await page.getByText("Dinner only").click();
    for (let i = 0; i < 4; i++) await page.getByLabel("Decrease days").click();
    await page.getByText("Generate Your Plan →").click();

    // URL should update to the real plan ID
    await expect(page).toHaveURL(/\/plan\/test-plan-uuid-123/, { timeout: 5000 });
  });
});

**tests/e2e/generation-error.spec.ts** (replace stub):

```typescript
import { test, expect } from "@playwright/test";

test.describe("Generation error handling", () => {
  test.beforeEach(async ({ page }) => {
    // Intercept mealPlan.create to return a valid plan ID
    await page.route("**/trpc/mealPlan.create*", async (route) => {
      await route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify({ result: { data: { id: "error-test-plan-456" } } }),
      });
    });

    await page.goto("/plan/new");
  });

  test("StreamErrorBanner shown when generate-draft returns 500", async ({ page }) => {
    // Route returns server error — no SSE body
    await page.route("**/functions/v1/generate-draft", async (route) => {
      await route.fulfill({ status: 500, body: "Internal Server Error" });
    });

    await page.getByText("Generate Your Plan →").click();

    await expect(
      page.getByText("Something went wrong while generating your plan.")
    ).toBeVisible({ timeout: 5000 });
  });

  test("partial meal cards remain visible after mid-stream error", async ({ page }) => {
    // Route returns 2 meals then aborts
    const twoMeals = [
      { day_of_week: "Monday", meal_type: "dinner", title: "Roast Chicken", short_description: "Classic roast." },
      { day_of_week: "Tuesday", meal_type: "dinner", title: "Pasta Bolognese", short_description: "Rich sauce." },
    ];
    const partialSSE = twoMeals.map((m) => `data: ${JSON.stringify(m)}`).join("\n\n") + "\n\n";

    await page.route("**/functions/v1/generate-draft", async (route) => {
      // Return partial SSE then close without [DONE] — simulates mid-stream failure
      await route.fulfill({
        status: 200,
        headers: { "Content-Type": "text/event-stream", "Cache-Control": "no-cache" },
        body: partialSSE,
      });
    });

    await page.getByText("Dinner only").click();
    for (let i = 0; i < 4; i++) await page.getByLabel("Decrease days").click();
    await page.getByText("Generate Your Plan →").click();

    // Partial cards should be visible
    await expect(page.getByText("Roast Chicken")).toBeVisible({ timeout: 5000 });
    await expect(page.getByText("Pasta Bolognese")).toBeVisible({ timeout: 5000 });
  });

  test("Try again button resets to GenerationForm state", async ({ page }) => {
    await page.route("**/functions/v1/generate-draft", async (route) => {
      await route.fulfill({ status: 500, body: "error" });
    });

    await page.getByText("Generate Your Plan →").click();

    await expect(
      page.getByText("Something went wrong while generating your plan.")
    ).toBeVisible({ timeout: 5000 });

    await page.getByText("Try again →").click();

    // Form should be visible again
    await expect(page.getByText("Generate Your Plan →")).toBeVisible({ timeout: 3000 });
  });
});
```
