Create `src/hooks/use-llm-logs.ts`:

```typescript
import { useQuery } from "@tanstack/react-query";
import { trpcClient } from "@/lib/trpc/client";

export type LlmLog = {
  id: string;
  model: string;
  prompt: string;
  response: string;
  prompt_tokens: number | null;
  completion_tokens: number | null;
  household_id: string | null;
  created_at: string;
};

export function useLlmLogs() {
  const query = useQuery<LlmLog[]>({
    queryKey: ["llm-logs"],
    queryFn: () => trpcClient.query("devTools.llmLogs") as Promise<LlmLog[]>,
    staleTime: 30_000,
  });

  return {
    logs: query.data ?? [],
    isLoading: query.isLoading,
    error: query.error instanceof Error ? query.error : null,
  };
}
```

```tsx
import { useState } from "react";
import { useLlmLogs, type LlmLog } from "@/hooks/use-llm-logs";

function formatTimestamp(iso: string): string {
  return new Date(iso).toLocaleString();
}

function formatTokens(log: LlmLog): string {
  const prompt = log.prompt_tokens ?? 0;
  const completion = log.completion_tokens ?? 0;
  return `${prompt + completion} tokens (${prompt}p + ${completion}c)`;
}

function LlmLogRow({ log }: { log: LlmLog }) {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div className="rounded-[1.5rem] bg-white/60 px-4 py-4 md:px-5">
      <div className="flex flex-wrap items-center gap-x-4 gap-y-1">
        <span className="text-sm font-semibold text-[var(--color-ink)]">{log.model}</span>
        <span className="text-xs text-[var(--color-muted)]">{formatTokens(log)}</span>
        <span className="text-xs text-[var(--color-muted)]">{formatTimestamp(log.created_at)}</span>
      </div>
      <div className="mt-2">
        <button
          type="button"
          onClick={() => setIsOpen((o) => !o)}
          aria-expanded={isOpen}
          className="min-h-[44px] px-2 text-sm text-[var(--color-sage-deep)] hover:underline"
        >
          {isOpen ? "Hide" : "Show prompt/response"}
        </button>
        {isOpen && (
          <div className="mt-2 space-y-2">
            <pre className="overflow-x-auto whitespace-pre-wrap break-words rounded-lg bg-white/60 p-3 text-xs leading-5 text-[var(--color-muted)]">
              {log.prompt}
            </pre>
            <pre className="overflow-x-auto whitespace-pre-wrap break-words rounded-lg bg-white/60 p-3 text-xs leading-5 text-[var(--color-muted)]">
              {log.response}
            </pre>
          </div>
        )}
      </div>
    </div>
  );
}

function LLMRequestsSection() {
  const { logs, isLoading } = useLlmLogs();

  return (
    <div className="rounded-[1.75rem] bg-[rgba(255,255,255,0.72)] px-6 py-6 shadow-[var(--shadow-soft)] backdrop-blur-sm md:px-8 md:py-8">
      <p className="text-xs uppercase tracking-[0.26em] text-[var(--color-muted)]">
        Developer tools
      </p>
      <h2 className="mt-3 font-display text-[20px] font-semibold leading-snug text-[var(--color-sage-deep)]">
        LLM Requests
      </h2>
      <p className="mt-2 text-sm leading-7 text-[var(--color-muted)]">
        Last 10 generation calls. Expand any entry to inspect the full prompt and response.
      </p>

      <div className="mt-6 space-y-3">
        {isLoading && (
          <p className="text-sm text-[var(--color-muted)]">Loading...</p>
        )}
        {!isLoading && logs.length === 0 && (
          <div className="rounded-xl bg-white/60 px-4 py-4 text-sm leading-7 text-[var(--color-muted)]">
            No LLM calls yet. Generate a plan to see logs here.
          </div>
        )}
        {logs.map((log) => (
          <LlmLogRow key={log.id} log={log} />
        ))}
      </div>
    </div>
  );
}

function SpoonacularUsageSection() {
  return (
    <div className="rounded-[1.75rem] bg-[rgba(255,255,255,0.72)] px-6 py-6 shadow-[var(--shadow-soft)] backdrop-blur-sm md:px-8 md:py-8">

      </p>
    </div>
  );
}

export function DevPage() {
```
