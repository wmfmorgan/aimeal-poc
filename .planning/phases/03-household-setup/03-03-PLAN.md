    ```typescript
    /**
     * use-household.ts
     *
     * TanStack Query hook for the household domain.
     * Exposes:
     *   - household: the current user's household (null = first-time user)
     *   - isLoading: query loading state
     *   - upsert: mutation for household.upsert (create or update)
     *
     * Pattern: mirrors src/hooks/use-ping-status.ts (useQuery + trpcClient).
     */

    import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
    import { trpcClient } from "@/lib/trpc/client";

    // ---------------------------------------------------------------------------
    // Types matching household.get return shape from the tRPC router
    // ---------------------------------------------------------------------------

    export type HouseholdMember = {
      id: string;
      name: string;
      allergies: string[];
      avoidances: string[];
      dietType: string;
    };

    export type Household = {
      id: string;
      name: string;
      cookingSkillLevel: string;
      appliances: string[];
      members: HouseholdMember[];
    };

    export type HouseholdUpsertInput = {
      name: string;
      cookingSkillLevel: "beginner" | "intermediate" | "advanced";
      appliances: string[];
      members: Array<{
        id?: string;
        name: string;
        allergies: string[];
        avoidances: string[];
        dietType?: string;
      }>;
    };

    // ---------------------------------------------------------------------------
    // Hook
    // ---------------------------------------------------------------------------

    export function useHousehold() {
      const queryClient = useQueryClient();

      const { data: household, isLoading, error } = useQuery<Household | null>({
        queryKey: ["household"],
        queryFn: () => trpcClient.query("household.get") as Promise<Household | null>,
        staleTime: 60_000,
      });

      const upsert = useMutation<{ id: string }, Error, HouseholdUpsertInput>({
        mutationFn: (input: HouseholdUpsertInput) =>
          trpcClient.mutation("household.upsert", input) as Promise<{ id: string }>,
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ["household"] });
        },
      });

      return {
        household: household ?? null,
        isLoading,
        error,
        upsert,
      };
    }
    ```

    ```typescript
      e.preventDefault();
      const errors = validateHousehold(householdName, members);

      setIsSubmitting(true);

      } catch {
        setSubmitError(HOUSEHOLD_COPY.errorBanner);

        setIsSubmitting(false);
    ```

    ```typescript
    function memberSummary(member: MemberDraft): string {
      const constraints: string[] = [];
      if (member.dietType) constraints.push(member.dietType);
      constraints.push(...member.allergies.slice(0, 3 - (member.dietType ? 1 : 0)));
      if (constraints.length === 0) return HOUSEHOLD_COPY.noConstraints;
      const shown = constraints.slice(0, 3);
      const remaining = member.allergies.length + (member.dietType ? 1 : 0) - shown.length;
      return shown.join(" · ") + (remaining > 0 ? ` +${remaining} more` : "");
    }
    ```
