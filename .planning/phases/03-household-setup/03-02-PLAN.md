      ts: number;

    Read the current `supabase/functions/trpc/index.ts` (44 lines). Rewrite it completely to add auth context and the household router namespace. ALL existing procedures (`ping`, `generateDraft`) must be preserved exactly.

    The complete rewritten file:

    ```typescript
    /**
     * tRPC router running inside a Supabase Edge Function (Deno).
     *
     * Uses @trpc/server fetch adapter — no Node.js required.
     * All imports use npm: specifiers per Deno 2 convention.
     */
    import { initTRPC, TRPCError } from "npm:@trpc/server@11";
    import { fetchRequestHandler } from "npm:@trpc/server@11/adapters/fetch";
    import { z } from "npm:zod@3";
    import { createClient } from "npm:@supabase/supabase-js@2";

    // ---------------------------------------------------------------------------
    // Auth context
    // ---------------------------------------------------------------------------

    type Context = {
      userId: string | null;
      supabase: ReturnType<typeof createClient>;
    };

    const t = initTRPC.context<Context>().create();

    /** Middleware: rejects unauthenticated callers with UNAUTHORIZED. */
    const authedProcedure = t.procedure.use(({ ctx, next }) => {
      if (!ctx.userId) throw new TRPCError({ code: "UNAUTHORIZED" });
      return next({ ctx: { ...ctx, userId: ctx.userId as string } });
    });

    // ---------------------------------------------------------------------------
    // Zod schemas
    // ---------------------------------------------------------------------------

    const householdUpsertInput = z.object({
      name: z.string().min(1),
      cookingSkillLevel: z.enum(["beginner", "intermediate", "advanced"]),
      appliances: z.array(z.string()),
      members: z.array(z.object({
        id: z.string().uuid().optional(),
        name: z.string().min(1),
        allergies: z.array(z.string()),
        avoidances: z.array(z.string()),
        dietType: z.string().optional(),
      })),
    });

    // ---------------------------------------------------------------------------
    // Router
    // ---------------------------------------------------------------------------

    const appRouter = t.router({
      ping: t.procedure.query(() => ({ pong: true, ts: Date.now() })),

      generateDraft: t.procedure
        .input(
          z.object({
            householdId: z.string().uuid(),
            numDays: z.number().int().min(1).max(14).default(7),
          })
        )
        .mutation(({ input }) => {
          // Stub — real implementation calls generate-draft edge function
          return {
            mealPlanId: crypto.randomUUID(),
            householdId: input.householdId,
            numDays: input.numDays,
            status: "draft",
            message: "tRPC → Edge Function wiring confirmed",
          };
        }),

      household: t.router({
        /**
         * Get the current user's household (with members).
         * Returns null if no household exists yet (first-time user).
         */
        get: authedProcedure.query(async ({ ctx }) => {
          const { data: household, error } = await ctx.supabase
            .from("households")
            .select("*")
            .eq("user_id", ctx.userId)
            .maybeSingle();

          if (error) throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: error.message });
          if (!household) return null;

          const { data: members, error: membersError } = await ctx.supabase
            .from("household_members")
            .select("id, name, allergies, avoidances, diet_type")
            .eq("household_id", household.id)
            .order("created_at", { ascending: true });

          if (membersError) throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: membersError.message });

          return {
            id: household.id as string,
            name: household.name as string,
            cookingSkillLevel: household.cooking_skill_level as string,
            appliances: (household.appliances ?? []) as string[],
            members: (members ?? []).map((m) => ({
              id: m.id as string,
              name: m.name as string,
              allergies: (m.allergies ?? []) as string[],
              avoidances: (m.avoidances ?? []) as string[],
              dietType: (m.diet_type ?? "") as string,
            })),
          };
        }),

        /**
         * Create or update the current user's household plus all members.
         *
         * Upsert strategy (safe for the UNIQUE constraint on user_id):
         *   1. Upsert the households row using ON CONFLICT (user_id) DO UPDATE.
         *      This NEVER deletes and recreates — safe because household_members
         *      have a cascade-delete FK on household_id.
         *   2. Load the current member IDs for this household from the DB.
         *   3. Delete members whose IDs are NOT in the incoming payload.
         *   4. Upsert each member in the payload (INSERT or UPDATE by id).
         */
        upsert: authedProcedure
          .input(householdUpsertInput)
          .mutation(async ({ ctx, input }) => {
            // Step 1: Upsert household row
            const { data: household, error: upsertError } = await ctx.supabase
              .from("households")
              .upsert(
                {
                  user_id: ctx.userId,
                  name: input.name,
                  cooking_skill_level: input.cookingSkillLevel,
                  appliances: input.appliances,
                },
                { onConflict: "user_id" }
              )
              .select("id")
              .single();

            if (upsertError) throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: upsertError.message });

            const householdId = household.id as string;

            // Step 2: Load existing member IDs
            const { data: existingMembers } = await ctx.supabase
              .from("household_members")
              .select("id")
              .eq("household_id", householdId);

            const existingIds = (existingMembers ?? []).map((m) => m.id as string);

            // Step 3: Delete removed members (IDs in DB but not in payload)
            const incomingIds = input.members
              .filter((m) => m.id !== undefined)
              .map((m) => m.id as string);

            const toDelete = existingIds.filter((id) => !incomingIds.includes(id));
            if (toDelete.length > 0) {
              const { error: deleteError } = await ctx.supabase
                .from("household_members")
                .delete()
                .in("id", toDelete);
              if (deleteError) throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: deleteError.message });
            }

            // Step 4: Upsert each member in the payload
            for (const member of input.members) {
              const memberRow = {
                household_id: householdId,
                name: member.name,
                allergies: member.allergies,
                avoidances: member.avoidances,
                diet_type: member.dietType ?? null,
                dietary_preferences: [] as string[],
              };

              if (member.id) {
                // Update existing member
                const { error } = await ctx.supabase
                  .from("household_members")
                  .update(memberRow)
                  .eq("id", member.id);
                if (error) throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: error.message });
              } else {
                // Insert new member
                const { error } = await ctx.supabase
                  .from("household_members")
                  .insert(memberRow);
                if (error) throw new TRPCError({ code: "INTERNAL_SERVER_ERROR", message: error.message });
              }
            }

            return { id: householdId };
          }),
      }),
    });

    export type AppRouter = typeof appRouter;

    // ---------------------------------------------------------------------------
    // Serve — inject auth context from Authorization header
    // ---------------------------------------------------------------------------

    Deno.serve(async (req) => {
      const authHeader = req.headers.get("Authorization") ?? "";
      const supabaseAdmin = createClient(
        Deno.env.get("SUPABASE_URL") ?? "",
        Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? "",
      );

      let userId: string | null = null;
      if (authHeader.startsWith("Bearer ")) {
        const token = authHeader.slice(7);
        const { data } = await supabaseAdmin.auth.getUser(token);
        userId = data.user?.id ?? null;
      }

      return fetchRequestHandler({
        endpoint: "/trpc",
        req,
        router: appRouter,
        createContext: () => ({ userId, supabase: supabaseAdmin }),
      });
    });
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → tRPC client → Edge Function | Authorization: Bearer token crosses this boundary. Token may be absent (unauthenticated) or forged (spoofing attempt). |
| Edge Function → Supabase DB (service role) | Service role bypasses RLS. The authedProcedure middleware is the only access gate. |
| tRPC input (Zod) | User-supplied strings (name, allergies, appliances) cross this boundary. Zod enforces shape before DB writes. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-01 | Spoofing | household.get / household.upsert | mitigate | `authedProcedure` middleware rejects requests with `userId === null`. Supabase `auth.getUser(token)` validates the JWT cryptographically — no decode-only verification. |
| T-3-02 | Tampering | household.upsert input | mitigate | Zod `householdUpsertInput` schema validates: `name` min(1), `cookingSkillLevel` enum, `appliances` string array, member `name` min(1), `allergies/avoidances` string arrays. Malformed input returns 400 before any DB write. |
| T-3-03 | Elevation of Privilege | household.get — user reads another user's household | mitigate | Query filters by `user_id: ctx.userId`. RLS policy `households: own rows` enforces `auth.uid() = user_id` at the DB level as defense-in-depth (service role is used but userId filter is explicit). |
| T-3-04 | Elevation of Privilege | household.upsert — user writes to another user's household | mitigate | Upsert uses `user_id: ctx.userId` as the conflict target — users can only upsert their own row. Member queries scope to `household_id` which was fetched for `ctx.userId`. |
| T-3-05 | Tampering | households table — duplicate rows via race condition | mitigate | UNIQUE constraint `households_user_id_key` (added in Plan 01) makes duplicate rows a DB error. Upsert uses `ON CONFLICT (user_id) DO UPDATE` — concurrent saves from the same user update the same row rather than creating a second one. |
| T-3-06 | Information Disclosure | SUPABASE_SERVICE_ROLE_KEY exposed in client | accept | Key lives in `supabase/functions/.env` (not in `src/` or committed to git). Edge Function environment — never sent to browser. Local dev risk only. |
</threat_model>
